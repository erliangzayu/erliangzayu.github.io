// 随机事件判断（最终版：默认全persist，仅persist:false例外，不影响其他随机逻辑） function checkRandomEvent(oc, now) { if (!oc.randomEvents || oc.randomEvents.length === 0) { return { isRandom: false, schedule: null }; } // 筛选当前时段符合条件的随机事件 const validEvents = oc.randomEvents.filter(event => { if (!event.week.includes(now.week)) return false; const [start, end] = event.timeRange.split("-"); const [startH, startM] = start.split(":").map(Number); const [endH, endM] = end.split(":").map(Number); const nowMinutes = now.hour * 60 + now.minute; const startMinutes = startH * 60 + startM; const endMinutes = endH * 60 + endM; if (endMinutes < startMinutes) { return nowMinutes >= startMinutes || nowMinutes <= endMinutes; } else { return nowMinutes >= startMinutes && nowMinutes <= endMinutes; } }); if (validEvents.length === 0) { return { isRandom: false, schedule: null }; } // ========== 核心修改：默认全persist，仅persist:false例外 ========== const needPersist = validEvents[0].persist !== false; // 默认true，仅persist:false才例外 let cacheKey = ""; if (needPersist) { // 自动生成缓存ID（OC名+周几+时间范围+当天日期） const today = new Date().toISOString().split('T')[0]; // 2026-03-10 const eventWeek = validEvents[0].week.join('_'); // 1（周一） const eventTime = validEvents[0].timeRange.replace(':', '').replace('-', '_'); // 0100_0130 cacheKey = `persist_random_${oc.name}_${eventWeek}_${eventTime}_${today}`; // 检查是否在事件时段内（缓存是否有效） const [eventStart, eventEnd] = validEvents[0].timeRange.split("-"); const [esH, esM] = eventStart.split(":").map(Number); const [eeH, eeM] = eventEnd.split(":").map(Number); const eventStartMinutes = esH * 60 + esM; const eventEndMinutes = eeH * 60 + eeM; const nowMinutes = now.hour * 60 + now.minute; let isInEventTime = false; if (eventEndMinutes < eventStartMinutes) { isInEventTime = nowMinutes >= eventStartMinutes || nowMinutes <= eventEndMinutes; } else { isInEventTime = nowMinutes >= eventStartMinutes && nowMinutes <= eventEndMinutes; } // 不在事件时段 → 清空缓存 if (!isInEventTime) { localStorage.removeItem(cacheKey); return { isRandom: false, schedule: null }; } // 检查localStorage缓存（持久化，刷新/关浏览器都在） const cachedResult = localStorage.getItem(cacheKey); if (cachedResult) { const result = JSON.parse(cachedResult); if (result.isRandom) showTip(result.schedule.tip); return result; } } else { // 仅persist:false的事件：走原有sessionStorage逻辑 const today = new Date().toLocaleDateString(); cacheKey = `random_event_${oc.name}_${today}_${now.hour}`; const cachedResult = sessionStorage.getItem(cacheKey); if (cachedResult) { const result = JSON.parse(cachedResult); if (result.isRandom) showTip(result.schedule.tip); return result; } } // ========== 通用随机判定逻辑 ========== const randomEvent = validEvents[Math.floor(Math.random() * validEvents.length)]; const randomNum = Math.floor(Math.random() * 100) + 1; const isTriggered = randomNum <= randomEvent.probability; const result = { isRandom: isTriggered, schedule: isTriggered ? randomEvent : null }; // 按类型存入不同存储 if (needPersist) { localStorage.setItem(cacheKey, JSON.stringify(result)); // 持久化 } else { sessionStorage.setItem(cacheKey, JSON.stringify(result)); // 会话级 } if (isTriggered) { showTip(randomEvent.tip); } return result; }
